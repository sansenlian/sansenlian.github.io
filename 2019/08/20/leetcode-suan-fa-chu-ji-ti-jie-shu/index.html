0. 二叉树的构建

如果大家要在本地跑代码可以看下这里，需要先建立二叉树，可以按照先序遍历的方式，只不过遍历的时候创建树而已，树的结构：

function TreeNode(val) {
    this.val = val;
    this.left = this.right = null;
}

层序遍历创建二叉树

不是用递归，用的是加入队列并不断判断队列长度来决定是否结束

/*
* 层序方式建立树,这也是最常见的
* @params {arr} arr[] 传入的数组
* @return {TreeNode} node 返回根节点
* */
function levelOderCreateTree(arr) {
    let i = 0;
    let node = new TreeNode();
    if (arr[i] == undefined || arr[i] == null) return node;
    node.val = arr[i];
    let queue = [];
    if (node !== null) {
        queue.push(node);
        i++;
    }
    while (queue.length != 0) {
        let len = queue.length;
        for (let j = 0; j < len; j++) {
            let currentNode = queue.shift();
            currentNode.left = new TreeNode();
            currentNode.right = new TreeNode();
            // console.log(currentNode.val,len)
            if (arr[i] != undefined && arr[i] !== null) {
                currentNode.left.val = arr[i];
                queue.push(currentNode.left);
            }
            i++;
            if (arr[i] != undefined && arr[i] !== null) {
                currentNode.right.val = arr[i];
                queue.push(currentNode.right);
            }
            i++;
        }
    }
    return node;
}
let root = levelOderCreateTree(arr);

1. 二叉树的最大深度

题目

题目详情

给定一个二叉树，找出其最大深度。

二叉树的深度为根节点到最远叶子节点的最长路径上的节点数。

说明: 叶子节点是指没有子节点的节点。

示例：
给定二叉树 [3,9,20,null,null,15,7]，

3
   / \
  9  20
    /  \
   15   7

题解

找深度，最容易想到的就是直接用层序遍历，但是函数要求返回一个number

所以用不递归

层序遍历应该写的非常熟练
信手拈来的感觉

var maxDepth = function(root) {
        if(root==null) return 0;
        let queue =[];
        queue.push(root);
        let level = 0;
        while(queue.length){
            level ++;
            let len = queue.length;
            for(let j=0;j<len;j++){
                let node = queue.shift();
                node.left && queue.push(node.left)
                node.right && queue.push(node.right);
            }
        }
        return level;
    };

2. 验证二叉搜索树

题目

题目详情

给定一个二叉树，判断其是否是一个有效的二叉搜索树。

假设一个二叉搜索树具有如下特征：

  • 节点的左子树只包含小于当前节点的数。
  • 节点的右子树只包含大于当前节点的数。
  • 所有左子树和右子树自身必须也是二叉搜索树。

示例 1:

输入:
    2
   / \
  1   3
输出: true

示例 2:

输入:
    5
   / \
  1   4
     / \
    3   6
输出: false
解释: 输入为: [5,1,4,null,null,3,6]。
     根节点的值为 5 ，但是其右子节点值为 4 。

题解

二叉搜索树中序遍历一定是有序

var isValidBST = function(root) {
        let res = [];
        function search(root){
            if(root==null || root.val==undefined)return;
            // left
            root.left!=null && root.left!=undefined && search(root.left);
            // mid
            res.push(root.val);
            // right
            root.right !=null && root.right!=undefined&& search(root.right);
        }
        search(root)
        // console.log(res)
        let len = res.length;
        for(let i = 0;i<len;i++){
            for(let j=i+1;j<len;j++){
                if(res[i]>=res[j])return false;
            }
        }
        return true;
    };

3. 对称二叉树

题目

题目详情

对称二叉树

给定一个二叉树，检查它是否是镜像对称的。

例如，二叉树 [1,2,2,3,4,4,3] 是对称的。

1
   / \
  2   2
 / \ / \
3  4 4  3

但是下面这个 [1,2,2,null,3,null,3] 则不是镜像对称的:

1
   / \
  2   2
   \   \
   3    3

说明:

如果你可以运用递归和迭代两种方法解决这个问题，会很加分。

题解


1. 递归

    var isSymmetric = function (root) {
        if (!root) return true;
        var Symmetric = function (p, q) {
            if (p == null && q != null) return false; // 一方面为空
            if (p != null && q == null) return false;   // 一方面为空
            if (p == null && q == null) return true; // 左右为空
            if (p.val != q.val) return false;       // 值不相等
            return Symmetric(p.left, q.right) && Symmetric(p.right, q.left); // 按照图中画的来比较
        };
        return Symmetric(root.left, root.right);
    };

2. 迭代

用一个栈来保存，模拟层序遍历

    var isSymmetric = function (root) {
        if (!root) return true;
        let s=[];
        s.push(root.left);
        s.push(root.right);
        while(s.length){
            let right = s.pop();
            let left = s.pop();
            if(right==null && left==null)continue;
            if(right==null || left==null) return false;
            if(right.val != left.val)return false;
            s.push(left.left);
            s.push(right.right);
            s.push(left.right);
            s.push(right.left);
        }
        return true;
    };

4. 二叉树的层次遍历

https://leetcode-cn.com/explore/interview/card/top-interview-questions-easy/7/trees/50/

题目详情

给定一个二叉树，返回其按层次遍历的节点值。 （即逐层地，从左到右访问所有节点）。
例如:
给定二叉树: [3,9,20,null,null,15,7],

        3
   / \
  9  20
    /  \
   15   7

返回其层次遍历结果：

[
  [3],
  [9,20],
  [15,7]
]

题解

1. 非递归写法

    var levelOrder = function(root) {
        if(!root)return [];
        let queue = [];
        queue.push(root);
        let res = [];
        while(queue.length){
            let level = [];
            var len = queue.length;
            for (var i = 0; i < len; i ++) {
                var currentNode = queue.shift();
                level.push(currentNode.val);
                if (currentNode.left !== null && currentNode.left.val!=undefined) queue.push(currentNode.left);
                if (currentNode.right !== null && currentNode.right.val!=undefined) queue.push(currentNode.right);
            }
            res.push(level);
        }
        return res;
    };

2.递归写法

怎么写递归呢，很多时候是把while循环里面的给解开成为递归，我们发现是要递归遍历每一层，那么每一层还是可以通过刚刚的非递归的方式来实现

    var levelOrder = function(root) {
        if(!root)return [];
        let queue = [],res=[];
        queue.push(root);
        function recircle(q){
            if(!q.length)return;
            let level = [];
            var len = queue.length;
            for (var i = 0; i < len; i ++) {
                var currentNode = queue.shift();
                level.push(currentNode.val);
                if (currentNode.left !== null && currentNode.left.val!=undefined) queue.push(currentNode.left);
                if (currentNode.right !== null && currentNode.right.val!=undefined) queue.push(currentNode.right);
            }
            res.push(level)
            recircle(queue)
        }
        recircle(queue)
        return res;
    };

5. 将所有有序数组转换为二叉树

题目详情

https://leetcode-cn.com/explore/interview/card/top-interview-questions-easy/7/trees/51/
将一个按照升序排列的有序数组，转换为一棵高度平衡二叉搜索树。

本题中，一个高度平衡二叉树是指一个二叉树每个节点 的左右两个子树的高度差的绝对值不超过 1。

示例:

给定有序数组: [-10,-3,0,5,9],

一个可能的答案是：[0,-3,9,-10,null,5]，它可以表示下面这个高度平衡二叉搜索树：

      0
     / \
   -3   9
   /   /
 -10  5

题解

其实就二分建立树

    var sortedArrayToBST = function(nums) {
        if(!nums)return [];
        function foo(left,right){
            if(left>right) return null; 
            let mid = left+(right-left>>1); // 如果是除法就要转为整数
            //console.log(right,left,nums[mid],mid)
            let _node = new TreeNode(nums[mid]);
            _node.left = foo(left,mid-1);
            _node.right = foo(mid+1,right);
            return _node;
        }
        return foo(0,nums.length-1); // 这里都是length-1，如果用slice处理就是直接length
    };
